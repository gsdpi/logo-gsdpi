N  = 100
n  = np.arange(N); 
c  = 1;
th = n*2*np.pi/360*137.5; 
r  = c*np.sqrt(n); 
x  = r*np.cos(th); 
y  = r*np.sin(th); 
t  = n/N
colors = hsv_to_rgb(np.column_stack((t*0.5+.3,0*t + .25,0*t + .95)))
plt.figure(figsize=(7,7)) 
plt.scatter(x,-y,s=(2-t)*450,c=colors,alpha=1)
